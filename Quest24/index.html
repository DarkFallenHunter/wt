Цепочка прототивов
var animal = {
  eats: true,
  sleeps: true
}

var dog = {
  bark: 'Woof-woof',
  wagTheTail: true,
  __proto__: animal
}

Object.defineProperty(dog, 'bark', {
  enumerable: false
})

var labrador = {
  name: 'John',
  bark: 'Woof-woof-woof',
  __proto__: dog
}

Object.defineProperty(labrador, 'name', {
  enumerable: false
})
